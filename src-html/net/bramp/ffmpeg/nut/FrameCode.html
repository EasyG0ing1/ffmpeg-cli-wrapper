001package net.bramp.ffmpeg.nut;
002
003import com.google.common.base.MoreObjects;
004
005public class FrameCode {
006
007  long flags;
008  int streamId;
009  int dataSizeMul;
010  int dataSizeLsb;
011  long ptsDelta;
012  int reservedCount;
013  long matchTimeDelta;
014  int headerIdx;
015
016  @Override
017  public String toString() {
018    return MoreObjects.toStringHelper(this)
019        .add("flags", flags)
020        .add("id", streamId)
021        .add("dataSizeMul", dataSizeMul)
022        .add("dataSizeLsb", dataSizeLsb)
023        .add("ptsDelta", ptsDelta)
024        .add("reservedCount", reservedCount)
025        .add("matchTimeDelta", matchTimeDelta)
026        .add("headerIdx", headerIdx)
027        .toString();
028  }
029}
